<div id="players"></div>

<script>
const requiredPlays = 4;
const playerCount = 30;
let players = [];

// Initialize players
for (let i = 1; i <= playerCount; i++) {
players.push({ name: `Player ${i}`, plays: 0, inGame: false });
}

function renderPlayers() {
const container = document.getElementById('players');
container.innerHTML = '';

players.forEach((p, index) => {
const div = document.createElement('div');
div.className = 'player';
if (p.inGame) div.classList.add('in');
if (p.plays >= requiredPlays) div.classList.add('done');

div.innerHTML = `
<strong>${p.name}</strong>
<span>Plays: ${p.plays}</span>
<div>
<button onclick="toggleInGame(${index})">${p.inGame ? 'Sub Out' : 'Sub In'}</button>
<button onclick="addPlay(${index})">+1 Play</button>
</div>
`;

container.appendChild(div);
});
}

function toggleInGame(index) {
players[index].inGame = !players[index].inGame;
renderPlayers();
}

function addPlay(index) {
if (players[index].inGame) {
players[index].plays++;
renderPlayers();
}
}

renderPlayers();
</script>
